```jsx
// Switcher breakdown for this category
const switcherAllocation = useMemo(() => {
  if (!categoryDetailData?.events) return [];
  const grouped = {};
  categoryDetailData.events.forEach(d => {
    const name = d.switcher || 'Unknown';
    grouped[name] = (grouped[name] || 0) + d.minutes / 60;
  });
  return Object.entries(grouped)
    .map(([name, hours]) => ({ name, hours: parseFloat(hours.toFixed(1)) }))
    .sort((a, b) => b.hours - a.hours);
}, [categoryDetailData]);

// Client distribution for this category
const clientDistribution = useMemo(() => {
  if (!categoryDetailData?.events) return [];
  const grouped = {};
  categoryDetailData.events.forEach(d => {
    const name = d.client || 'Unknown';
    grouped[name] = (grouped[name] || 0) + d.minutes / 60;
  });
  return Object.entries(grouped)
    .map(([name, value]) => ({ name, value: parseFloat(value.toFixed(1)) }))
    .sort((a, b) => b.value - a.value);
}, [categoryDetailData]);

// Trend data for this category (hours per week)
const categoryTrendData = useMemo(() => {
  if (!categoryDetailData?.events) return [];
  const byWeek = {};
  categoryDetailData.events.forEach(d => {
    const weekStart = new Date(d.dateObj);
    weekStart.setDate(weekStart.getDate() - weekStart.getDay() + 1); // Monday
    const key = weekStart.toISOString().slice(0, 10);
    byWeek[key] = (byWeek[key] || 0) + d.minutes / 60;
  });
  return Object.entries(byWeek)
    .map(([date, hours]) => ({ date, hours: parseFloat(hours.toFixed(1)) }))
    .sort((a, b) => a.date.localeCompare(b.date));
}, [categoryDetailData]);
```

const clientCategoryBreakdown = useMemo(() => {
  if (type !== 'switcher' || !data) return [];
  const byClient = {};
  data.forEach(d => {
    const client = d.client || 'Unknown';
    if (!byClient[client]) byClient[client] = { client, categories: {}, total: 0 };
    const cat = d.categoryName || 'Misc';
    byClient[client].categories[cat] = (byClient[client].categories[cat] || 0) + d.minutes / 60;

  return Object.values(byClient)

      categories: Object.entries(item.categories)
