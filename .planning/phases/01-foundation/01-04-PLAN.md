import { describe, it, expect } from 'vitest';
import { parseCSV } from '../parseCSV.js';

describe('parseCSV', () => {
  it('parses valid CSV with all expected fields', () => {
    const csv = `switcher,date,department,client,task category,task,time spent
Luke,14/01/2026,Marketing,Acme Corp,Strategy Meeting,Q1 planning,60`;
    const result = parseCSV(csv);
    expect(result).toHaveLength(1);
    expect(result[0].switcher).toBe('Luke');
    expect(result[0].client).toBe('Acme Corp');
    expect(result[0].minutes).toBe(60);
    expect(result[0].dateObj).toBeInstanceOf(Date);
    expect(result[0].dateObj.getFullYear()).toBe(2026);
    expect(result[0].dateObj.getMonth()).toBe(0); // January = 0
    expect(result[0].dateObj.getDate()).toBe(14);
  });

  it('filters rows with invalid dates instead of using today', () => {
    const csv = `switcher,date,client,time spent
Luke,not-a-date,Acme,30`;
    const result = parseCSV(csv);
    expect(result).toHaveLength(0);
  });

  it('returns empty array for empty input', () => {
    expect(parseCSV('')).toHaveLength(0);
    expect(parseCSV('   \n  ')).toHaveLength(0);
  });

  it('returns empty array for headers-only CSV', () => {
    const csv = `switcher,date,client,time spent`;
    const result = parseCSV(csv);
    expect(result).toHaveLength(0);
  });

  it('filters rows missing required fields', () => {
    const csv = `switcher,date,client,time spent
,14/01/2026,Acme,60
Luke,14/01/2026,,60`;
    const result = parseCSV(csv);
    expect(result).toHaveLength(0);
  });

  it('parses time spent as integer minutes', () => {
    const csv = `switcher,date,client,time spent
Luke,14/01/2026,Acme,90`;
    const result = parseCSV(csv);
    expect(result[0].minutes).toBe(90);
  });

  it('sorts output by date ascending', () => {
    const csv = `switcher,date,client,time spent
Luke,20/01/2026,Acme,60
Luke,10/01/2026,Acme,30
Luke,15/01/2026,Acme,45`;
    const result = parseCSV(csv);
    expect(result[0].dateObj.getDate()).toBe(10);
    expect(result[1].dateObj.getDate()).toBe(15);
    expect(result[2].dateObj.getDate()).toBe(20);
  });

  it('handles quoted commas in field values', () => {
    const csv = `switcher,date,client,time spent
Luke,14/01/2026,"Acme, Inc",60`;
    const result = parseCSV(csv);
    expect(result[0].client).toBe('Acme, Inc');
  });

  it('parses dd/mm/yyyy date format correctly', () => {
    const csv = `switcher,date,client,time spent
Luke,05/12/2025,Acme,60`;
    const result = parseCSV(csv);
    // 05/12/2025 = December 5, 2025
    expect(result[0].dateObj.getMonth()).toBe(11); // December = 11
    expect(result[0].dateObj.getDate()).toBe(5);
    expect(result[0].dateObj.getFullYear()).toBe(2025);
  });
});

import { describe, it, expect } from 'vitest';
import { getWeekNumber } from '../getWeekNumber.js';

describe('getWeekNumber', () => {
  it('returns week 1 for January 1, 2026 (Thursday)', () => {
    // Jan 1, 2026 is a Thursday — ISO week 1
    expect(getWeekNumber(new Date(2026, 0, 1))).toBe(1);
  });

  it('returns correct week for a known mid-year date', () => {
    // July 1, 2026 is a Wednesday — should be week 27
    const week = getWeekNumber(new Date(2026, 6, 1));
    expect(week).toBeGreaterThan(25);
    expect(week).toBeLessThan(29);
  });

  it('handles year boundary correctly', () => {
    // Dec 31, 2025 is a Wednesday — ISO week 1 of 2026
    const week = getWeekNumber(new Date(2025, 11, 31));
    expect(week).toBe(1);
  });

  it('returns consistent results for same date', () => {
    const date = new Date(2026, 3, 15);
    const first = getWeekNumber(date);
    const second = getWeekNumber(date);
    expect(first).toBe(second);
  });

  it('returns week number between 1 and 53', () => {
    // Test 12 dates spread across the year
    for (let month = 0; month < 12; month++) {
      const week = getWeekNumber(new Date(2026, month, 15));
      expect(week).toBeGreaterThanOrEqual(1);
      expect(week).toBeLessThanOrEqual(53);
    }
  });
});

**src/shared/utils/__tests__/aggregation.test.js:**

import { describe, it, expect } from 'vitest';
import { buildTrendData } from '../aggregation.js';

// Helper to create test data entries
const makeEntry = (dateStr, minutes) => {
  const parts = dateStr.split('/');
  return {
    dateStr,
    dateObj: new Date(parts[2], parts[1] - 1, parts[0]),
    minutes,
  };
};

describe('buildTrendData', () => {
  it('groups by day correctly', () => {
    const data = [
      makeEntry('14/01/2026', 60),
      makeEntry('14/01/2026', 30),
      makeEntry('15/01/2026', 45),
    ];
    const result = buildTrendData(data, 'day');
    // Should have 2 buckets: Jan 14 and Jan 15
    expect(result.length).toBe(2);
  });

  it('sums hours within each bucket', () => {
    const data = [
      makeEntry('14/01/2026', 60),
      makeEntry('14/01/2026', 30),
    ];
    const result = buildTrendData(data, 'day');
    // 60 + 30 = 90 minutes = 1.5 hours
    const bucket = result[0];
    expect(bucket.hours).toBeCloseTo(1.5, 1);
  });

  it('returns empty array for empty input', () => {
    expect(buildTrendData([], 'day')).toHaveLength(0);
  });

  it('groups by month correctly', () => {
    const data = [
      makeEntry('14/01/2026', 60),
      makeEntry('20/01/2026', 60),
      makeEntry('14/02/2026', 60),
    ];
    const result = buildTrendData(data, 'month');
    // Should have 2 month buckets: Jan and Feb
    expect(result.length).toBe(2);
  });

  it('returns results sorted chronologically', () => {
    const data = [
      makeEntry('15/02/2026', 60),
      makeEntry('10/01/2026', 60),
    ];
    const result = buildTrendData(data, 'day');
    // First bucket should be January, second February
    // Verify chronological order
    expect(result.length).toBe(2);
  });
});
